  <script>
    const { createApp } = Vue;
    const app = createApp({
      data() {
        return {
          // 启动页相关
          skipText: '跳过 3s',
          countdown: 3,
          
          // 登录相关
          phone: '',
          code: '',
          password: '',
          isPwdLogin: false,
          
          // 注册相关
          regPhone: '',
          regCode: '',
          regPassword: '',
          
          // 保存穿搭相关
          outfitName: '',
          outfitPrice: '',
          
          // 设置相关
          notifyEnabled: true,
          vibrateEnabled: false,
          showContentEnabled: true,
          nickname: '刀刀',
          
          // 密码重置相关
          oldPwd: '',
          newPwd: '',
          confirmPwd: ''
        };
      },
      mounted() {
        // 启动页倒计时
        const timer = setInterval(() => {
          this.countdown--;
          this.skipText = `跳过 ${this.countdown}s`;
          if (this.countdown <= 0) {
            clearInterval(timer);
            this.goToLogin();
          }
        }, 1000);
      },
      methods: {
        // 显示指定页面
        showPage(pageId) {
          document.querySelectorAll('.page').forEach(page => {
            page.classList.remove('active');
          });
          document.getElementById(pageId).classList.add('active');
          
          // 更新底部导航激活状态
          const tabItems = document.querySelectorAll('.tabbar-item');
          tabItems.forEach(item => {
            item.classList.remove('active');
          });
          const activeTab = document.querySelector(`.tabbar-item[onclick="app.showPage('${pageId}')"]`);
          if (activeTab) activeTab.classList.add('active');
        },
        
        // 跳转登录页
        goToLogin() {
          this.showPage('loginPage');
        },
        
        // 跳转注册页
        goToRegister() {
          this.showPage('registerPage');
        },
        
        // 切换登录方式
        toggleLoginType() {
          this.isPwdLogin = !this.isPwdLogin;
        },
        
        // 获取验证码
        getCode() {
          if (!this.phone) {
            vant.Toast('请输入手机号码');
            return;
          }
          vant.Toast.success('验证码已发送');
        },
        
        // 获取注册验证码
        getRegCode() {
          if (!this.regPhone) {
            vant.Toast('请输入手机号码');
            return;
          }
          vant.Toast.success('验证码已发送');
        },
        
        // 登录
        login() {
          if (!this.phone) {
            vant.Toast('请输入手机号码');
            return;
          }
          if (this.isPwdLogin && !this.password) {
            vant.Toast('请输入密码');
            return;
          }
          if (!this.isPwdLogin && !this.code) {
            vant.Toast('请输入验证码');
            return;
          }
          this.showPage('homePage');
        },
        
        // 注册
        register() {
          if (!this.regPhone) {
            vant.Toast('请输入手机号码');
            return;
          }
          if (!this.regCode) {
            vant.Toast('请输入验证码');
            return;
          }
          if (!this.regPassword) {
            vant.Toast('请输入密码');
            return;
          }
          vant.Toast.success('注册成功');
          this.showPage('loginPage');
        },
        
        // 跳转智能衣橱页
        goToSmartWardrobe() {
          this.showPage('smartWardrobePage');
        },
        
        // 跳转服装详情页
        goToClothesDetail() {
          this.showPage('clothesDetailPage');
        },
        
        // 显示筛选弹窗
        showFilter() {
          vant.Toast('筛选功能待实现');
        },
        
        // 显示删除弹窗
        showDelete() {
          vant.Toast('删除功能待实现');
        },
        
        // 收藏服装
        collectClothes() {
          vant.Toast.success('收藏成功');
        },
        
        // 保存穿搭
        saveOutfit() {
          if (!this.outfitName) {
            vant.Toast('请输入穿搭名称');
            return;
          }
          vant.Toast.success('保存成功');
          this.showPage('wardrobePage');
        },
        
        // 跳转个人信息页
        goToUserInfo() {
          this.showPage('userInfoPage');
        },
        
        // 跳转我的收藏页
        goToCollection() {
          this.showPage('collectionPage');
        },
        
        // 跳转设置页
        goToSettings() {
          this.showPage('settingsPage');
        },
        
        // 跳转通知页
        goToNotification() {
          this.showPage('notificationPage');
        },
        
        // 跳转字体页
        goToFont() {
          this.showPage('fontPage');
        },
        
        // 跳转语言页
        goToLanguage() {
          this.showPage('languagePage');
        },
        
        // 跳转重设密码页
        goToResetPwd() {
          this.showPage('resetPwdPage');
        },
        
        // 更换头像
        changeAvatar() {
          vant.ActionSheet.create({
            actions: [
              { name: '拍照' },
              { name: '从相册选取' }
            ],
            onSelect: (action) => {
              vant.Toast(action.name + '功能待实现');
            }
          }).show();
        },
        
        // 重设密码
        resetPassword() {
          if (!this.oldPwd) {
            vant.Toast('请输入原密码');
            return;
          }
          if (!this.newPwd) {
            vant.Toast('请输入新密码');
            return;
          }
          if (this.newPwd !== this.confirmPwd) {
            vant.Toast('两次密码输入不一致');
            return;
          }
          vant.Toast.success('密码重置成功');
          this.showPage('settingsPage');
        },
        
        // 退出登录
        logout() {
          vant.Dialog.confirm({
            title: '提示',
            message: '确定要退出登录吗？'
          }).then(() => {
            this.showPage('loginPage');
          });
        }
      }
    }).mount('#app');

    // 暴露app到全局，用于模板字符串中的onclick调用
    window.app = app;
  </script>
